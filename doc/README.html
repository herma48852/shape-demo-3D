3D Shapes Demo

This project provides a real-time visualization client allowing developers to observe, publish, and subscribe to geometric spatial telemetry topics over an active DDS network. It is fully backwards-compatible with RTI's legacy 2D Shapes Demo, rendering flat 2D shape publications with a sleek, uniform extrusion depth on the 3D grid.

Architectural Overview

To move out of the browser sandbox and link straight against the native RTI Connext DDS 7.7 libraries, we implement a decoupled multithreaded pattern. In standard DDS systems, receiving messages occurs on thread context pools owned exclusively by the middleware. If those threads attempt to issue direct drawing calls to the window graphics context, the application context crashes.

DDS Thread Context (Input)

Runs standard asynchronous UDP multicast discovery and network packet listener loops. Receives raw binary frames from the wire, deserializes them into type-safe IDL shapes, and enqueues them onto a thread-safe lock-free buffer.

UI Render Thread Context (Main)

Monitors keyboard/mouse callback contexts, processes local kinematics, flushes queue frames at screen refresh rates (60+ Hz), and writes flat primitive buffers straight to standard unlit OpenGL pipelines.

RTI 2D Shapes Demo Backwards Compatibility

Fully compatible with standard RTI 2D Shapes Demo publications. When subscribing to 2D topics (Z=125), the visualizer automatically extrudes the 2D plane data to a crisp 3D thickness of 4 units, rendering beautiful, depth-shaded cylinders and prisms.

Matches the exact schema of your custom shapes trajectory definition, featuring extensible structure extensions (`ShapeTypeExtended`) inheritance alongside fill configurations.

    SOLID_FILL,
    TRANSPARENT_FILL,
    HORIZONTAL_HATCH_FILL,
    VERTICAL_HATCH_FILL
};

struct ShapeType
{
    string<128> color; //@key
    long x;
    long y;
    long shapesize;
};//@Extensibility EXTENSIBLE_EXTENSIBILITY 

struct ShapeTypeExtended : ShapeType {

3 Thread-Safe Telemetry Queue (thread_queue.hpp)

Connects both thread contexts securely. To handle type extensibility smoothly, our payload queue encapsulates the full extensible schema structure `ShapeTypeExtended`. 

#pragma once
#include <mutex>
#include <vector>
#include "ShapeType.hpp" // Fully resolved type mappings output by rtiddsgen

class ThreadSafeSampleQueue {
private:
    std::mutex mutex_;
    std::vector<ShapeTypeExtended> pending_;

public:
    void push_sample(const ShapeTypeExtended& sample) {
        std::lock_guard<std::mutex> lock(mutex_);
        pending_.push_back(sample);
    }

    std::vector<ShapeTypeExtended> drain_queue() {
        std::lock_guard<std::mutex> lock(mutex_);
        std::vector<ShapeTypeExtended> drained = std::move(pending_);
        pending_.clear();
        return drained;
    }
};

4 DDS Infrastructure Subsystem manager (dds_manager.cpp)

Encapsulates your compiled Connext layer. Initializes type-safe modern C++ Participant and Topic instances and sets up DataReaders to receive shapes. 

#include "dds_manager.hpp"
#include <iostream>

void DdsSubsystemManager::initialize_dds_domain(int32_t domain_id, ThreadSafeSampleQueue* target_queue) {
    target_queue_ = target_queue;

    try {
        // 1. Create DomainParticipant bound to standard default QoS
        participant_ = dds::domain::DomainParticipant(domain_id);

        // 2. Register Topic schema using the type-safe generated class ShapeTypeExtended
        topic_ = dds::topic::Topic<ShapeTypeExtended>(participant_, "Sphere");

        // 3. Instantiate publisher and subscriber elements
        publisher_ = dds::pub::Publisher(participant_);
        subscriber_ = dds::sub::Subscriber(participant_);

        // 4. Create local DataReader with Reliable Quality of Service
        dds::sub::qos::DataReaderQos reader_qos;
        reader_qos << dds::core::policy::Reliability::Reliable();
        
        reader_ = dds::sub::DataReader<ShapeTypeExtended>(subscriber_, topic_, reader_qos);

        // 5. Initialize the asynchronous read dispatcher loop
        std::cout << "RTI Connext DDS 7.7 Participant initialized on Domain: " << domain_id << std::endl;
        
        // Non-blocking wait loop pattern
        while (true) {
            auto samples = reader_.take();
            for (const auto& sample : samples) {
                if (sample.info().valid()) {
                    target_queue_->push_sample(sample.data());
                }
            }
            std::this_thread::sleep_for(std::chrono::milliseconds(10));
        }
    } catch (const std::exception& ex) {
        std::cerr << "RTI Connext DDS 7.7 Initialization Exception: " << ex.what() << std::endl;
    }
}

void DdsSubsystemManager::publish_local_shape(const std::string& color, int32_t x, int32_t y, int32_t size) {
    dds::pub::DataWriter<ShapeTypeExtended> writer(publisher_, topic_);

    ShapeTypeExtended sample;
    sample.color(color);
    sample.x(x);
    sample.y(y);
    sample.shapesize(size);
    sample.fillKind(ShapeFillKind::SOLID_FILL);
    sample.angle(0.0f);

    writer.write(sample);
}

5 Unlit Rendering Engine Context (graphics_engine.cpp)

Initializes GLFW window settings and renders simple, unlit shape primitives (squares and circles) with custom thickness, avoiding lighting artifacts. 

#include "graphics_engine.hpp"
#include <GLFW/glfw3.h>

void NativeGraphicsEngine::initialize_viewport() {
    glfwInit();
    window_ = glfwCreateWindow(1280, 720, "RTI 3D Shapes Workspace", NULL, NULL);
    glfwMakeContextCurrent(window_);
    glfwSwapInterval(1); // Lock draw calls synchronously to V-Sync frame rates

    // Initialize clean projection matrices
    glMatrixMode(GL_PROJECTION);
    glLoadIdentity();
    glOrtho(0, 250, 0, 250, -500, 500);
    glMatrixMode(GL_MODELVIEW);
}

void NativeGraphicsEngine::draw_planar_square(float x, float y, float z, float size, float r, float g, float b) {
    glPushMatrix();
    glTranslatef(x, y, z);
    
    // Constant solid color with unlit basic shaders
    glColor3f(r, g, b);
    float half = size / 2.0f;

    // Draw square faces parallel to Z=125 grid slice
    glBegin(GL_QUADS);
        glVertex3f(-half, -half, 0.0f);
        glVertex3f(half, -half, 0.0f);
        glVertex3f(half, half, 0.0f);
        glVertex3f(-half, half, 0.0f);
    glEnd();

    glPopMatrix();
}

6 Application Orchestration & Threads Entry (main.cpp)

Connects all layers. Boots up GLFW/OpenGL, instantiates your concurrent DDS managers, and drains the thread-safe queue on every frame refresh. 

#include <iostream>
#include <thread>
#include "thread_queue.hpp"
#include "dds_manager.hpp"
#include "graphics_engine.hpp"

int main(int argc, char* argv[]) {
    std::cout << "Booting RTI native workspace client..." << std::endl;

    // 1. Initialize thread-safe synchronization queue boundary
    ThreadSafeSampleQueue sample_queue;

    // 2. Initialize DDS Subsystem manager on independent background execution context
    DdsSubsystemManager dds_manager;
    std::thread dds_thread([&]() {
        dds_manager.initialize_dds_domain(0, &sample_queue);
    });

    // 3. Initialize Graphics Engine context
    NativeGraphicsEngine gfx_engine;
    gfx_engine.initialize_viewport();

    // 4. Synchronous Application Drawing loop
    while (!glfwWindowShouldClose(gfx_engine.get_window())) {
        glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
        glLoadIdentity();

        // Drain pending queue samples received asynchronously from UDP wire
        auto samples = sample_queue.drain_queue();
        for (const auto& sample : samples) {
            // Unpack ShapeTypeExtended elements cleanly via C++11 modern value-type bindings
            float r = (sample.color() == "BLUE") ? 0.2f : 0.9f;
            float g = (sample.color() == "GREEN") ? 0.9f : 0.2f;
            float b = (sample.color() == "BLUE") ? 0.9f : 0.2f;

            gfx_engine.draw_planar_square(
                static_cast<float>(sample.x()), 
                static_cast<float>(sample.y()), 
                125.0f, // Locked planar Z-coordinate
                static_cast<float>(sample.shapesize()), 
                r, g, b
            );
        }

        glfwSwapBuffers(gfx_engine.get_window());
        glfwPollEvents();
        std::this_thread::sleep_for(std::chrono::milliseconds(16)); // Yield ~60Hz
    }

    // Clean shutdown orchestration
    dds_thread.join();
    return 0;
}
